    <script>
        const apiKey = ""; 

        const companies = [
            { id: 1, name: "Amazonas Logística", icon: "📦", desc: "Salaris precaris i ritmes frenètics al magatzem." },
            { id: 2, name: "Hotel Paradís", icon: "🏨", desc: "Kellys explotades i molta temporalitat estacional." },

            { id: 9, name: "Menjadors Escolars SA", icon: "🍎", desc: "Pocs minuts per dinar i càrrega física extrema." }
        ];

        const actions = [
            { id: 'f', name: "Full Informatiu", cost: 4, val: 4, icon: "📄", desc: "Piquet a l'entrada per explicar els drets." },
            { id: 'c', name: "Cafè clandestí", cost: 6, val: 7, icon: "☕", desc: "Reunió discreta amb els treballadors indecisos." },
            { id: 'i', name: "Denúncia Inspecció", cost: 12, val: 12, icon: "⚖️", desc: "Cop legal per irregularitats greus." },
            { id: 'v', name: "Vaga Parcial", cost: 22, val: 20, icon: "📢", desc: "Aturada d'una hora per pressionar RRHH." }
        ];

        const obstacles = [
            { name: "Campanya de por", impact: 6, msg: "L'empresa amenaça amb acomiadaments si hi ha massa mobilització." },
            { name: "Pizzes de 'bon rotllo'", impact: 4, msg: "RRHH regala pizzes per apaivagar els ànims de la plantilla." },
            { name: "El Sindicat Groc", impact: 8, msg: "Un sindicat afavorit per la direcció intoxica amb notícies falses." },
            { name: "Promeses buides", impact: 5, msg: "La gerència promet una pujada salarial que mai arriba al paper." },
            { name: "Fatiga laboral", impact: 3, msg: "L'excés d'hores extres fa que la gent passi de sindicats." }
        ];

        let state = {
            company: null,
            week: 1,
            credit: 20,
            maxCredit: 20,
            support: 5,
            iaUsed: false
        };

        function init() {
            const grid = document.getElementById('company-grid');
            companies.forEach(c => {
                const div = document.createElement('div');
                div.className = "selectable-card card-collage p-6 text-center";
                div.innerHTML = `
                    <div class="text-5xl mb-4">${c.icon}</div>
                    <div class="font-black uppercase text-lg">${c.name}</div>
                    <p class="text-[11px] mt-2 italic leading-tight text-zinc-600">${c.desc}</p>
                `;
                div.onclick = () => {
                    state.company = c;
                    document.querySelectorAll('.selectable-card').forEach(x => x.classList.remove('selected'));
                    div.classList.add('selected');
                };
                grid.appendChild(div);
            });
            renderActions();
        }

        function checkSetup() {
            if (!state.company) return alert("Si us plau, tria una empresa per començar la lluita!");
            document.getElementById('screen-setup').classList.remove('active');
            document.getElementById('screen-game').classList.add('active');
            document.getElementById('display-company').innerText = state.company.name;
            updateUI();
        }

        function renderActions() {
            const grid = document.getElementById('actions-grid');
            grid.innerHTML = '';
            actions.forEach(a => {
                const can = state.credit >= a.cost;
                const div = document.createElement('div');
                div.className = `card-collage p-4 flex flex-col justify-between ${!can ? 'opacity-40 grayscale' : ''}`;
                div.innerHTML = `
                    <div><h4 class="titular text-xl">${a.icon} ${a.name}</h4><p class="text-xs italic">${a.desc}</p></div>
                    <button onclick="doAction('${a.id}')" ${!can ? 'disabled' : ''} class="btn-sindical mt-4 py-2 text-xs">EXECUTAR (${a.cost}H)</button>
                `;
                grid.appendChild(div);
            });
        }

        function doAction(id) {
            const a = actions.find(x => x.id === id);
            if (state.credit < a.cost) return;
            state.credit -= a.cost;
            state.support += a.val;
            addLog(`✅ S'ha executat: ${a.name} (+${a.val}% de vots)`);
            updateUI();
        }

        function updateUI() {
            document.getElementById('stat-credit').innerText = `${state.credit}/${state.maxCredit}H`;
            document.getElementById('bar-credit').style.width = `${(state.credit/state.maxCredit)*100}%`;
            document.getElementById('stat-support').innerText = `${state.support}%`;
            document.getElementById('bar-support').style.width = `${Math.min(100, state.support)}%`;
            document.getElementById('current-week').innerText = state.week;
            renderActions();
        }

        function nextTurn() {
            if (state.week >= 12) return endGame();
            state.week++;
            state.credit = state.maxCredit;
            state.iaUsed = false;
            
            // Increment de la dificultat: desgast natural
            state.support = Math.max(0, state.support - 2);
            addLog(`📉 Desgast: Els vots baixen un 2% per la inacció setmanal.`);

            // Traves de l'empresa (Event aleatori d'oposició)
            if (Math.random() > 0.5) {
                const obs = obstacles[Math.floor(Math.random() * obstacles.length)];
                state.support = Math.max(0, state.support - obs.impact);
                addLog(`⚠️ ${obs.name.toUpperCase()}: ${obs.msg} (-${obs.impact}% vots)`);
            }
            
            addLog(`La setmana ${state.week} ha començat.`);
            updateUI();
        }

        function addLog(m) {
            const l = document.getElementById('log-content');
            const d = document.createElement('div');
            d.className = "mb-2 border-l-4 border-black pl-2 py-1 bg-white";
            d.innerHTML = `<span class="opacity-60 text-[10px] font-bold uppercase">SET ${state.week}:</span><br>${m}`;
            l.prepend(d);
        }

        async function callGemini(prompt, system) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            
            let retries = 0;
            const maxRetries = 5;
            
            while (retries < maxRetries) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            contents: [{ parts: [{ text: prompt }] }],
                            systemInstruction: { parts: [{ text: system }] }
                        })
                    });
                    const data = await response.json();
                    return data.candidates?.[0]?.content?.parts?.[0]?.text;
                } catch (e) {
                    retries++;
                    await new Promise(resolve => setTimeout(resolve, Math.pow(2, retries) * 1000));
                }
            }
            return "S'ha produït un error de connexió sindical. L'empresa ha tallat el Wi-Fi?";
        }

        async function openIA(type) {
            if (state.iaUsed && (type === 'discurs' || type === 'proposta')) {
                return alert("Només pots fer una acció de campanya IA per setmana!");
            }
            
            const modal = document.getElementById('gemini-modal');
            const body = document.getElementById('modal-body');
            const footer = document.getElementById('ia-result-actions');
            const loader = document.getElementById('ia-loading');
            
            modal.classList.add('active');
            body.innerText = "";
            footer.innerHTML = "";
            loader.classList.remove('hidden');

            let prompt, system, title;
            const commonContext = `Empresa: ${state.company.name}. Sector: ${state.company.desc}. Setmana: ${state.week}/12. Suport actual: ${state.support}%.`;

            switch(type) {
                case 'discurs':
                    title = "📢 DISCURS ÈPIC";
                    system = "Ets un líder sindical carismàtic. Fes discursos breus, motivadors i un punt agressius contra RRHH. Utilitza un català correcte però popular. Evita paraules en castellà.";
                    prompt = `${commonContext}\nRedacta un discurs de 45 paraules per convèncer els treballadors que ens votin.`;
                    break;
                case 'proposta':
                    title = "📜 PROGRAMA ELECTORAL";
                    system = "Ets un expert en negociació de convenis. Les teves propostes són realistes però ambicioses per a la classe treballadora. En català.";
                    prompt = `${commonContext}\nCrea 3 punts de programa electoral realistes per a aquesta empresa específica.`;
                    break;
                case 'negociacio':
                    title = "📩 RECLAMACIÓ RRHH";
                    system = "Ets un advocat laboralista del sindicat. El teu to és formal, seriós i amenaçador legalment. En català.";
                    prompt = `${commonContext}\nEscriu una reclamació de 40 paraules per un incompliment flagrant de la normativa laboral.`;
                    break;
                case 'consell':
                    title = "🧠 ESTRATÈGIA IA";
                    system = "Ets un estratega d'eleccions sindicals amb molta experiència. Analitza la situació i dóna un consell clau en català.";
                    prompt = `${commonContext}\nDóna'm un consell ràpid i mordaç per guanyar vots aquesta setmana.`;
                    break;
            }

            document.getElementById('modal-title').innerText = title;
            
            const res = await callGemini(prompt, system);
            loader.classList.add('hidden');
            body.innerText = res;

            if (type === 'discurs') {
                footer.innerHTML = `<button onclick="applyIAEffect('boost', 8)" class="btn-sindical">PUBLICAR (+8% VOTS)</button>`;
            } else if (type === 'proposta') {
                footer.innerHTML = `<button onclick="applyIAEffect('boost', 5)" class="btn-sindical">DIFONDRE (+5% VOTS)</button>`;
            } else if (type === 'negociacio') {
                footer.innerHTML = `<button onclick="applyIAEffect('credit', 10)" class="btn-sindical">PRESENTAR (+10H CRÈDIT)</button>`;
            } else {
                footer.innerHTML = `<button onclick="closeIA()" class="btn-sindical">ENTÈS</button>`;
            }
        }

        function applyIAEffect(type, val) {
            if (type === 'boost') {
                state.support += val;
                addLog(`✨ IA: Discurs/Proposta llançada amb èxit! +${val}% vots.`);
                state.iaUsed = true;
            } else if (type === 'credit') {
                state.credit = Math.min(state.maxCredit, state.credit + val);
                addLog(`✨ IA: S'ha guanyat crèdit extra per via legal! +${val}H.`);
            }
            updateUI();
            closeIA();
        }

        function closeIA() { document.getElementById('gemini-modal').classList.remove('active'); }

        function endGame() {
            const win = state.support >= 51;
            document.body.innerHTML = `

                    <p class="mb-10 italic">A l'empresa ${state.company.name.toUpperCase()}</p>

Els drets dels treballadors estan fora de perill gràcies a la teva lluita.' : 'La direcció i el sindicat groc han guanyat aquesta vegada. La plantilla continuarà desprotegida, però la lluita segueix.'}</p>

            </div>`;
        }
